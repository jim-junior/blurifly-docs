import { ReactVideo } from "reactjs-media";

## Usage

To use the video componentyou have to import it and use it.

### Code

```jsx
import React from "react";
import { ReactVideo } from "reactjs-media";

const App = () => {
	return (
		<div>
			<ReactVideo src="/video.mp4" poster="poster.jpg" type="video/mp4" />
		</div>
	);
};

export default App;
```
